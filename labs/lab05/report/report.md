function hypocycloid(x, r0, n)
    # —Ä–∞–¥–∏—É—Å –º–∞–ª–æ–π –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏:
    r0 = r0
    # –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –¥–ª—è –ø–æ—Å—Ç—Ä–æ–µ–Ω–∏—è –±–æ–ª—å—à–æ–π –æ–∫—Ä—É–∂–Ω–æ—Å—Ç–∏:
    k = x
    # —á–∏—Å–ª–æ –æ—Ç—Å—á—ë—Ç–æ–≤:
    count = n
    # –º–∞—Å—Å–∏–≤ –∑–Ω–∞—á–µ–Ω–∏–π —É–≥–ª–∞ tetha:
    tetha = collect(0: 2*œÄ/100 : 10*œÄ+2*œÄ/count)
    
    # –º–∞—Å—Å–∏–≤—ã –∑–Ω–∞—á–µ–Ω–∏–π –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç:
    x_1 = r0 * k * cos.(tetha)
    y_1 = r0 * k * sin.(tetha)
    #–í –∫–æ–Ω—Ü–µ —Å–¥–µ–ª–∞–µ–º –∞–Ω–∏–º–∞—Ü–∏—é –ø–æ–ª—É—á–∏–≤—à–µ–≥–æ—Å—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
    anim = @animate for i in 1:count

        plt=plot(5, 
            xlim=(-k-1, k+1),
            ylim=(-k-1, k+1), 
            color=:red, 
            aspect_ratio=1,
            legend=false, 
            framestyle=:origin)
